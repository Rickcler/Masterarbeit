\section{General Notation}

Let  $(X_t)$ be a \textit{categorical}  time series process, where each $X_t$ takes values in

\[
S := \{s_0, \dots, s_m\}, \qquad m \in \mathbb{N}.
\]
$(X_t)$ may be \textit{nominal}, in which case the order of magnitude of the elements in $S$ is arbitrary and solely usesd for notational consistency, or $ordinal$, in which case the order of the elements in $S$ reflects the inherent order of the elements.
\section{Stationarity}
Furthermore, $(X_t)$ is assumed to be stationary. A distinction can be made between two forms of stationarity:
\begin{definition}[Weak Stationarity]
    A process $(X_t)$ is said to be \emph{weakly stationary} if:
        \[
            \begin{aligned}
            \mathrm{E}(X_t) &= \mu_X \quad &&\forall t, \\
            \mathrm{E}(X_t^2) &< \infty \quad &&\forall t, \\
            \mathrm{Cov}(X_t, X_{t+h}) &= \gamma(h) \quad &&\forall t,h.
            \end{aligned}
        \]
\end{definition}

\begin{definition}[Strong/ Strict Stationarity]
    Strong/ Strict stationarity requires that all finite-dimensional distributions are invariant under time shifts, i.e.,
        \begin{equation}
            F_{X_t,\ldots,X_{t+j}} = F_{X_{t+h},\ldots,X_{t+h+j}} \quad \forall\,t,j,h.
        \end{equation}
\end{definition}
\section{Marginal Distributions}

Regardless of whether we consider a simple categorical process with few categories or attempt to describe a complex, temporally evolving process using moments and measures of serial dependence, the analysis of categorical processes is always built on the examination of the marginal probabilities. However, because ordinal and nominal data encode different types of information, their marginals are typically represented in different ways:

\begin{itemize}
    \item \textbf{Ordinal data:}
    Most analytical tools for ordinal processes take into account the natural ordering of the elements in $S$. For this reason, most methods for ordinal data rely on the marginal cumulative distribution function (CDF), which accumulates probabilities in accordance with the natural ordering of the categories:
    \[
        \boldsymbol{f} = (f_0, \dots, f_{m-1})^\top,
        \qquad
        f_i := P(X_t \le s_i).
    \]

    For ordinal data, it is sufficient to consider the cumulative probabilities of the categories $0, \ldots , m-1$ since the cumulative probability of the last category $m$ is always equal to 1. \\

    Although one could equivalently estimate the PMF and derive the CDF from it, representing ordinal data through the CDF is often more convenient as it this form that is used to compute metrics for location and disperson.

    \item \textbf{Nominal data:}
    Since the categories of nominal variables do not possess a meaningful order, statistical procedures cannot make use of cumulative structure.
    Therefore the marginal distribution is best represented by the probability mass function (PMF):
    \[
        \boldsymbol{p} = (p_0, \dots, p_{m})^\top,
        \qquad
        p_i := P(X_t = s_i).
    \]
\end{itemize}


\section{Bivariate Probabilities at Lag $h$}

\begin{definition}[Bivariate lag-$h$ probabilities]
\label{def:biv_probs}
Let $(X_t)$ be a stationary categorical time series. The bivariate lag-$h$ 
probabilities are defined as follows.
\begin{enumerate}
    \item[\normalfont(i)] \textbf{Ordinal series:}
    \[
    f_{ij}(h) := P(X_t \le s_i,\, X_{t+h} \le s_j), \qquad i,j = 0,\ldots, m-1.
    \]
    \item[\normalfont(ii)] \textbf{Nominal series:}
    \[
    p_{ij}(h) := P(X_t = s_i,\, X_{t+h} = s_j), \qquad i,j = 0,\ldots, m.
    \]
\end{enumerate}
Under serial independence, the bivariate probabilities factorize as 
$f_{ij}(h) = f_i f_j$ and $p_{ij}(h) = p_ip_j$ respectively.
\end{definition}
\section{Location}

For a nominal process, any measure that relies on ordering is inappropriate.
Instead, a location measure must depend only on the category probabilities.
The mode, defined as the most probable category,
\[
\operatorname{Mod}(X_t) = \arg\max_{s_i \in S} P(X_t = s_i),
\]
is therefore the natural choice. It provides a meaningful summary of the process by identifying the most frequent category.\\
For an ordinal process, the natural ordering of the categories allows for additional measures of central tendency.
While the mode can still be informative as the most frequent category, the median respects the ordering of categories and is often more useful for describing the central location:
\[
\operatorname{Med}(X_t)
=
\min \bigl\{\, s_i \in S : F(s_i) \ge \tfrac{1}{2} \,\bigr\}.
\]
Thus, for ordinal data, both the mode and the median carry information, but they capture slightly different aspects of location: the mode identifies the most typical category, whereas the median indicates the midpoint in the ordered scale.
\section{Measures of Dispersion}

Since the concept of dispersion differs between ordinal and nominal processes, two separate measures are employed: the \emph{Index of Ordinal Variation} (IOV) for ordinal data and the \emph{Index of Qualitative Variation} (IQV) for nominal data, defined as
\begin{equation}\label{eq:iov_iqv}
    \mathrm{IOV}(\boldsymbol{f}) = \frac{4}{m}\sum_{i=0}^{m-1}f_i(1-f_i), \qquad
    \mathrm{IQV}(\boldsymbol{p}) = \frac{m+1}{m}\!\left(1-\sum_{i=0}^{m}p_i^2\right).
\end{equation}
Both indices are scaled to $[0,1]$, with $0$ indicating a degenerate (one-point) distribution and $1$ indicating maximal dispersion. However, what constitutes maximal dispersion differs: for ordinal data, dispersion is maximal when all probability mass is concentrated in the two outermost categories, whereas for nominal data it corresponds to the uniform distribution over all $m+1$ categories.
\begin{proposition}[Dispersion Bounds]\label{prop:dispersion_bounds}
Both $\mathrm{IOV}$ and $\mathrm{IQV}$ satisfy:
\begin{enumerate}
    \item[\normalfont(i)] $\mathrm{IOV}=0$ (resp.\ $\mathrm{IQV}=0$) if and only if
    the distribution is degenerate.
    \item[\normalfont(ii)] $\mathrm{IOV}=1$ iff
    $\boldsymbol{f}=(\tfrac{1}{2},\ldots,\tfrac{1}{2})^\top$;
    $\mathrm{IQV}=1$ iff $p_i=\tfrac{1}{m+1}$ for all $i$.
\end{enumerate}
\end{proposition}
\begin{proof}
\textit{Part (i), ordinal case.} % dein bestehender Beweis fÃ¼r IOV=0
A one-point distribution assigns all probability mass to a single category
$s_j \in S$.
For ordinal data, this corresponds to a CDF vector $\mathbf{f}$ of the form
\[
\mathbf{f} \in
\left\{
\begin{pmatrix} 1 \\ 1 \\ \vdots \\ 1 \end{pmatrix},
\begin{pmatrix} 0 \\ 1 \\ \vdots \\ 1 \end{pmatrix},
\ldots,
\begin{pmatrix} 0 \\ 0 \\ \vdots \\ 1 \end{pmatrix}
\right\}
:= \{\boldsymbol{c}_0,\ldots,\boldsymbol{c}_m\} \subset [0,1]^{m+1}.
\]
In each of these cases, every component $f_i$ satisfies $f_i \in \{0,1\}$, and thus
\[
f_i(1 - f_i) = 0 \qquad \forall i.
\]
Therefore,
\[
\mathrm{IOV}
= \frac{4}{m}\sum_{i=0}^{m-1} f_i (1 - f_i)
= 0.
\]
Conversely, if $\mathrm{IOV}=0$, then
\[
\sum_{i=0}^{m-1} f_i(1 - f_i) = 0,
\]
and since each term is nonnegative,
\[
f_i(1 - f_i) \overset{!}{=} 0 \quad \forall i.
\]
This is only possible if $f_i \in \{0,1\}$ for all $i$, which implies
$\mathbf{f} \in \{\boldsymbol{e}_0,\ldots,\boldsymbol{e}_m\}$.
Hence, $\mathrm{IOV}=0$ holds \emph{exactly} for a degenerate one-point distribution.\\
\textit{Part (i), nominal case.} % dein bestehender Beweis fÃ¼r IQV=0
A degenerate (one--point) distribution assigns all probability mass to a single
category $s_j \in S$, which corresponds to a PMF vector of the form
\[
\mathbf{p} \in
\left\{
\begin{pmatrix} 1 \\ 0 \\ \vdots \\ 0 \end{pmatrix},
\begin{pmatrix} 0 \\ 1 \\ \vdots \\ 0 \end{pmatrix},
\ldots,
\begin{pmatrix} 0 \\ 0 \\ \vdots \\ 1 \end{pmatrix}
\right\}
:= \{\boldsymbol{e}_0,\ldots,\boldsymbol{e}_m\} \subset [0,1]^{m+1}.
\]
In each of these cases, exactly one component equals $1$ and all others equal $0$, so
\[
\sum_{i=0}^m p_i^2 = 1.
\]
Thus,
\[
\mathrm{IQV}
= \frac{m+1}{m}\bigl(1 - \sum_{i=0}^m p_i^2\bigr)
= \frac{m+1}{m}(1 - 1) = 0.
\]
Conversely, if $\mathrm{IQV} = 0$, then
\[
\frac{m+1}{m}\Bigl(1 - \sum_{i=0}^m p_i^2\Bigr) = 0
\quad\Longrightarrow\quad
\sum_{i=0}^m p_i^2 = 1.
\]
As all probabilities satisfy $0 \leq p_i \leq 1$, it follows that \[
p_i^2 \leq p_i \quad \forall p_i, \text{ and } p_i^2 < p_i \quad \forall p_i \notin \{0,1\}.
\]
Since $\sum_{i=1}^{m}p_i = 1$, it follows that the only way $\sum_i p_i^2 = 1$ can hold is if exactly one $p_i$ equals 1 and all others are 0. In other words, the distribution is degenerate and $\boldsymbol{p} \in \{\boldsymbol{e}_0,\ldots, \boldsymbol{e}_{m}\}$.

\textit{Part (ii), ordinal case.} % dein bestehender Beweis fÃ¼r IOV=1
Dispersion of an ordinal variable is maximal when half of the
probability mass lies in the lowest category and half in the highest
category; with all intermediate categories having probability zero.
This results in a CDF vector,

\[
\mathbf{f} = \left(\tfrac{1}{2}, \tfrac{1}{2}, \ldots, \tfrac{1}{2}\right)^\top.
\]
Then
\[
f_i(1 - f_i) = \frac{1}{2}\left(1 - \frac{1}{2}\right) = \frac{1}{4} \qquad \forall i,
\]
and thus
\[
\mathrm{IOV}
= \frac{4}{m} \cdot \sum_{i=0}^{m-1} \frac{1}{4}
= \frac{4}{m} \cdot \frac{m}{4}
= 1.
\]
To show that $\mathrm{IOV} = 1 \Rightarrow \boldsymbol{f} = (\tfrac{1}{2}, \ldots, \tfrac{1}{2})^T$, recall that the IOV only depends on $\boldsymbol{f}$ through the sum
\[
\sum_{i=0}^{m-1} f_i (1 - f_i).
\]
Taking the derivative of $g(f_i) := f_i(1-f_i)$  and setting it to zero yields:
\begin{align*}
\frac{\partial}{\partial f_i} g(f_i) &= 1 - 2 f_i, \\
1 - 2 f_i &\overset{!}{=} 0, \\
f_i &= \frac{1}{2}.
\end{align*}

Since $g(f_i=0) = g(f_i = 1)=0$, the fucntion is concave and this critical point is indeed the global maximum.
Since its been already shown that $\mathrm{IOV}(\boldsymbol{f} = (\frac{1}{2}, \ldots, \frac{1}{2})^T) = 1$, it directly follows that
\[
\mathrm{IOV}(\boldsymbol{f}) = 1 \Leftrightarrow \boldsymbol{f} = (\frac{1}{2}, \ldots, \frac{1}{2})^\top
\]

\textit{Part (ii), nominal case.} % dein bestehender Beweis fÃ¼r IQV=1
For nominal data, maximal dispersion is attained when probability mass is distributed uniformly across all $m+1$ categories, i.e.
\[
p_i = \frac{1}{m+1}, \qquad i = 0,\ldots,m.
\]
In this case,
\[
\sum_{i=0}^m p_i^2 = (m+1)\frac{1}{(m+1)^2} = \frac{1}{m+1},
\]
and hence
\[
\mathrm{IQV}
= \frac{m+1}{m}\left(1 - \frac{1}{m+1}\right)
= 1.
\]
Conversely, suppose that $\mathrm{IQV} = 1$. Then
\[
\frac{m+1}{m}\left(1 - \sum_{i=0}^m p_i^2\right) = 1
\quad\Longrightarrow\quad
\sum_{i=0}^m p_i^2 = \frac{1}{m+1}.
\]
Thus, maximizing $\mathrm{IQV}$ is equivalent to minimizing
\[
f(\boldsymbol p) := \sum_{i=0}^m p_i^2
\quad \text{subject to} \quad
\sum_{i=0}^m p_i = 1.
\]
Introducing the Lagrangian (with Lagrange multiplier $\lambda$)
\[
\mathcal L(\boldsymbol p, \lambda)
:= \sum_{i=0}^m p_i^2 - \lambda\left(\sum_{i=0}^m p_i - 1\right),
\]
the first-order conditions are
\begin{align*}
    \frac{\partial \mathcal L}{\partial p_j} = 2p_j - \lambda &= 0\\
    \Rightarrow \quad 2p_j &= \lambda, \qquad j=0,\dots,m,
\end{align*}
together with the constraint $\sum_{i=0}^m p_i = 1$.
Since the same multiplier $\lambda$ applies to all components, it follows that
\[
p_0 = p_1 = \dots = p_m.
\]
Imposing the constraint yields
\[
(m+1)p_j = 1 \quad \Longrightarrow \quad p_j = \frac{1}{m+1} \quad \forall j \in 0,\dots, m .
\]
Hence, $f(\boldsymbol p)$ is uniquely minimized at
\[
\boldsymbol p = \left(\tfrac{1}{m+1}, \dots, \tfrac{1}{m+1}\right)^\top,
\]
and therefore $\mathrm{IQV}$ attains its maximal value at the uniform distribution.

\end{proof}


\section{Measures of serial dependence}

The marginal distribution of a categorical time series captures the long-run 
frequencies of the individual categories, but contains no information about the 
temporal structure of the process. To describe how the process evolves over time, 
measures of serial dependence are required.\\
For continuous-valued processes, the autocorrelation function (ACF) is the standard 
tool for quantifying temporal dependence. However, the ACF is not directly applicable 
to categorical data, since it relies on the existence of meaningful arithmetic 
operations on the observed values. For nominal data, such operations are undefined, 
and for ordinal data they are quesitonable, as the categories carry no numerical meaning beyond their ordering.\\
Dependence measures for categorical time series are typically constructed 
from the bivariate lag-$h$ probabilities introduced in defintion \ref{def:biv_probs}. 
Under serial independence, these would factorize as $f_{ij}(h) = f_if_j$ and 
$p_{ij}(h) = p_ip_j$ respectively. Measures for serial dependence are naturally 
constructed by quantifying the deviation of the bivariate probabilities from this 
independence benchmark.\\
\begin{definition}[Cohen's $\kappa$]
\label{def:cohens-kappa}
Natural and widely used measures of serial dependence are the ordinal 
and nominal versions of Cohen's $\kappa$, which compare the joint 
probabilities to those expected under independence. For lag $h$, they 
are defined as
\[
    \kappa_{\mathrm{ord}}(h) 
    = \frac{\sum_{j=0}^{m-1}\bigl(f_{jj}(h) - f_j^2\bigr)}
           {\sum_{i=0}^{m-1} f_i(1-f_i)}
    \quad \text{and} \quad
    \kappa_{\mathrm{nom}}(h) 
    = \frac{\sum_{j=0}^{m}\bigl(p_{jj}(h) - p_j^2\bigr)}
           {1-\sum_{i=0}^{m} p_i^2}.
\]
\end{definition}
The numerator of each measure sums the diagonal deviations $f_{jj}(h) - f_j^2$
(resp.\ $p_{jj}(h) - p_j^2$), which capture the excess probability of observing
the same category at times $t$ and $t+h$ relative to what would be expected under
independence. The denominator normalizes the numerator by a measure of marginal dispersion:
for the ordinal version it equals $\frac{m}{4}\cdot\mathrm{IOV}$, and for the
nominal version it is equal to $\frac{m}{m+1}\cdot \mathrm{IQV}$. This ensures that $\kappa(h)$ is invariant to
the overall level of dispersion in the marginal distribution. For both measures, positive values indicate that the process tends to remain in the same
category across lag $h$, while negative values indicate a tendency to switch
categories.\\
Under perfect positive dependence, i.e.\ $P(X_{t+h} = s_i \mid X_t = s_i) = 1$
for all $s_i \in S$, we have $f_{jj}(h) = f_j$ and $p_{jj}(h) = p_j$, and both
measures attain their upper bound:
\begin{align*}
    \kappa_{\mathrm{ord}}(h) 
    &= \frac{\sum_{j=0}^{m-1}(f_j - f_j^2)}{\sum_{i=0}^{m-1}(f_i - f_i^2)} = 1, \\[1ex]
    \kappa_{\mathrm{nom}}(h) 
    &= \frac{\sum_{j=0}^{m}(p_j - p_j^2)}{1 - \sum_{i=0}^{m}p_i^2}
    = \frac{1 - \sum_{j=0}^{m}p_j^2}{1 - \sum_{i=0}^{m}p_i^2} = 1.
\end{align*}
Under serial independence, $f_{jj}(h) = f_j^2$ and $p_{jj}(h) = p_jp_j$, so that
the numerator vanishes and $\kappa(h) = 0$.\\
For the nominal version, the lower bound is attained when $p_{jj}(h) = 0$ for all
$j$, i.e.\ when the process never remains in the same category across lag $h$:
\[
\kappa_{\mathrm{nom}}(h) 
= \frac{-\sum_{j=0}^{m}p_j^2}{1 - \sum_{i=0}^{m}p_i^2},
\]
which depends on the marginal distribution and equals $-1$ only in the special case
of a uniform marginal. For the ordinal version, the lower bound takes an analogous
form but involves the cumulative probabilities $f_j$.

\section{Estimation of Marginal Distributions}
To estimate the marginal (cumulative) probabilities, we express the
relative frequencies through a binarization of the original process
$(X_t)$. For ordinal data, define the indicator vector
\[
\boldsymbol{Z}_t := (Z_{t,0}, \ldots, Z_{t,m-1})^\top,
\qquad
Z_{t,i} := \mathbb{I}\{X_t \le s_i\},
\]
and for nominal data analogously
\[
\boldsymbol{Y}_t := (Y_{t,0}, \ldots, Y_{t,m})^\top,
\qquad
Y_{t,i} := \mathbb{I}\{X_t = s_i\}.
\]
The marginal CDF and PMF vectors are estimated by the corresponding sample means:
\begin{definition}[Empirical CDF and PMF estimators]
\label{def:empirical_estimators}
\[
\hat{\boldsymbol{f}} := \frac{1}{n} \sum_{t=1}^n \boldsymbol{Z}_t,
\qquad
\hat{\boldsymbol{p}} := \frac{1}{n} \sum_{t=1}^n \boldsymbol{Y}_t.
\]
\end{definition}

\section{Asymptotics of the marginal estimator}
To derive the asymptotic distribution of $\hat{\boldsymbol{f}}$ and
$\hat{\boldsymbol{p}}$, a central limit theorem for dependent data is
required. Since independence is not assumed, we impose weak dependence
conditions on the original process $(X_t)$.\\
Specifically, $(X_t)_{t \in \mathbb{Z}}$ is assumed to be strictly
stationary and $\alpha$-mixing with exponentially decaying coefficients.\\
\begin{definition}[$\alpha$-Mixing with Exponential Decay]\label{def:mixing}
Following Bradley (2005), let
\[
\mathcal{F}_J^L := \sigma(X_t : J \le t \le L)
\]
denote the $\sigma$-field generated by $\{X_t : J \le t \le L\}$.
The $\alpha$-mixing coefficients of $(X_t)$, given that it's strictly stationary, are defined as
\[
\alpha_{X_t}(n)
:= \sup_{A \in \mathcal{F}_{-\infty}^0,\; B \in \mathcal{F}_{n}^{\infty}}
\bigl| P(A \cap B) - P(A)P(B) \bigr|.
\]
The process is called $\alpha$-mixing if $\alpha(n) \to 0$ as $n \to \infty$.\\
We additionally assume exponential decay of the mixing coefficients,
i.e., there exist constants $C > 0$ and $\rho \in (0,1)$ such that
\begin{equation}\label{eq:exp_mixing}
\alpha(n) \le C \rho^n \qquad \text{for all } n \in \mathbb{N}.
\end{equation}

\end{definition}
The indicator mappings
\[
g_i(X_t) := \mathbb{I}\{X_t \le s_i\},
\qquad
h_i(X_t) := \mathbb{I}\{X_t = s_i\},
\]
are measurable functions of $X_t$ (see Klenke, 2005, p.~36).
Since measurability preserves strict stationarity, the binarized processes
$(g_i(X_t))$ and $(h_i(X_t))$ are strictly stationary whenever $(X_t)$ is
strictly stationary (see Kallenberg, 2002, p.~558).
Consequently, the vector-valued processes $\boldsymbol{Z}_t$ and $\boldsymbol{Y}_t$
are strictly stationary as well.\\
Moreover, each indicator variable $Z_{t,i}$ is defined on the same probability space
as $X_t$, and the $\sigma$-field it generates satisfies
\[
\sigma(Z_{t,i} : J \le t \le L) \subset \sigma(X_t : J \le t \le L), \qquad J \le L,
\]
which implies $\alpha_{Z_{t,i}}(n) \le \alpha_{X_t}(n)$ for all $i$ and $n$. Hence,
$\boldsymbol{Z}_t$ and $\boldsymbol{Y}_t$ are also $\alpha$-mixing with exponentially
decaying coefficients.\\
To derive the asymptotic distribution of $\hat{\boldsymbol{f}}$, we apply the
Cramér--Wold device: $\hat{\boldsymbol{f}}$ converges in distribution to a multivariate
normal if and only if every scalar projection
\[
U_t := \boldsymbol{a}^{\top}\boldsymbol{Z}_t, \qquad 
\boldsymbol{a} \in \mathbb{R}^{m}\setminus\{\boldsymbol{0}\},
\]
satisfies a univariate CLT. To see that $U_t$ is measurable, let
\[
L: \mathbb{R}^{m+1} \rightarrow \mathbb{R}, \quad L(\boldsymbol{z}) = \boldsymbol{a}^{\top}\boldsymbol{z}.
\]
Since $L$ is linear on a finite-dimensional space, it is continuous 
and hence Borel-measurable \citep[Theorem~1.88]{klenke2005}. 
The measurability of $g(X_t) = \boldsymbol{Z}_t$ follows from the 
measurability of its components, and the composition 
$U_t = L \circ g \circ X_t$ is measurable by 
\citet[Theorem~1.80]{klenke2005}. Hence $U_t$ inherits strict stationarity and $\alpha$-mixing with
$\alpha_{U_t}(n) \le \alpha_{X_t}(n) \le C\rho^n$ from $(X_t)$. We apply the CLT of
Ibragimov (1962, Theorem~7), which requires four conditions:
\begin{enumerate}
    \item[\normalfont(i)] \textbf{Strict stationarity.} Verified above, as $U_t$ is
    a measurable function of the strictly stationary process $(X_t)$.
    
    \item[\normalfont(ii)] \textbf{$\alpha$-mixing.} Verified above, with
    $\alpha_{U_t}(n) \le C\rho^n$.
    
    \item[\normalfont(iii)] \textbf{Summability.} The mixing coefficients are required
    to satisfy
    \[
    \sum_{n=1}^{\infty} \alpha(n)^{\lambda/(2+\lambda)} < \infty
    \quad \text{for some } \lambda > 0.
    \]
    Under exponential decay $\alpha(n) \le C\rho^n$, this follows from
    \begin{align*}
    \sum_{n=1}^{\infty} \alpha(n)^{\lambda/(2+\lambda)}
    &\le C^{\lambda/(2+\lambda)} \sum_{n=1}^{\infty} \rho^{\,n\lambda/(2+\lambda)} \\
    &= C^{\lambda/(2+\lambda)} \frac{\rho^{\lambda/(2+\lambda)}}
    {1-\rho^{\lambda/(2+\lambda)}} < \infty,
    \end{align*}
    since $0 < \rho^{\lambda/(2+\lambda)} < 1$ for all $\lambda > 0$.
    
    \item[\normalfont(iv)] \textbf{Moment condition.} The theorem requires
    $\mathrm{E}|U_t|^{2+\lambda} < \infty$ for some $\lambda > 0$. Since each
    $Z_{t,i} \in \{0,1\}$, the projection $U_t$ is bounded by
    $M := \sum_{i=0}^{m-1}|a_i| < \infty$, and hence
    $\mathrm{E}|U_t|^{2+\lambda} \le M^{2+\lambda} < \infty$ for all $\lambda > 0$.
\end{enumerate}
All four conditions being satisfied, Ibragimov's CLT applies to $(U_t)$, and by the 
Cramér--Wold device, the asymptotic variance matrix $\boldsymbol{\Sigma}_Z = 
(\sigma_{Z;i,j})$ is determined by
\[
\boldsymbol{a}^{\top}\boldsymbol{\Sigma}_Z\boldsymbol{a} = \mathrm{Var}(U_t)
+ 2\sum_{h=1}^{\infty}\mathrm{Cov}(U_t, U_{t+h}),
\]
which gives
\[
\sigma_{Z;i,j} = \mathrm{Cov}(Z_{t,i}, Z_{t,j})
+ \sum_{h=1}^{\infty}\bigl[\mathrm{Cov}(Z_{t,i}, Z_{t+h,j}) 
+ \mathrm{Cov}(Z_{t,j}, Z_{t+h,i})\bigr].
\]
Using $\mathrm{Cov}(Z_{t,i}, Z_{t,j}) = f_{\min\{i,j\}} - f_if_j$ and
$\mathrm{Cov}(Z_{t,i}, Z_{t+h,j}) = f_{ij}(h) - f_if_j$, the elements of
$\boldsymbol{\Sigma}_Z$ take the explicit form
\[
\sigma_{Z;i,j} = \bigl(f_{\min\{i,j\}} - f_if_j\bigr)
+ \sum_{h=1}^{\infty}\bigl(f_{ij}(h) + f_{ji}(h) - 2f_if_j\bigr).
\]

\begin{theorem}[Asymptotic Normality of $\hat{\boldsymbol{f}}$]\label{thm:clt_fhat}
Let $(X_t)$ be strictly stationary and $\alpha$-mixing with exponentially decaying
coefficients. Then
\[
\sqrt{n}\bigl(\hat{\boldsymbol{f}} - \boldsymbol{f}\bigr)
\xrightarrow[n\to\infty]{d}
\mathcal{N}\bigl(\boldsymbol{0},\, \boldsymbol{\Sigma}_Z\bigr),
\]
where $\boldsymbol{\Sigma}_Z = (\sigma_{Z;i,j})_{i,j=0,\ldots,m-1}$ is given by
\begin{equation}\label{eq:sigma_Z}
\sigma_{Z;i,j} = \bigl(f_{\min\{i,j\}} - f_if_j\bigr)
+ \sum_{h=1}^{\infty}\bigl(f_{ij}(h) + f_{ji}(h) - 2f_if_j\bigr).
\end{equation}
\end{theorem}
\section{Asymptotic Properties of Sample Dispersion Measures}
\subsection*{IOV}
We can now now use this information of the asymptotic distribution of $\hat{\boldsymbol{f}}$ to compute the asymptotic properties of sample measures that are based on $\hat{\boldsymbol{f}}$. According to the multivariate delta-method a random variable $G^* := g^*(\boldsymbol{\Theta^*})$, where $\boldsymbol{\Theta^*} \in \mathbb{R}^{m}$, $\sqrt{n}(\boldsymbol{\Theta^*} - \boldsymbol{\Theta})\overset{asy.}{\sim} \mathcal{N}(0, \boldsymbol{\Sigma^*})$ and  $g^*: \mathbb{R}^m\rightarrow \mathbb{R}$ is a continously differentiable function, satisfies
\[\sqrt{n}(G^* - g^*(\boldsymbol{\Theta})) \overset{asy.}{\sim} \mathcal{N}(0, \boldsymbol{j\Sigma^* j^{\top}} )\]
where $\boldsymbol{j}$ is the one-dimensional Jacobian-matrix (i.e. vector) and where $j_i =  \frac{\partial g^*(\boldsymbol{\Theta})}{\partial \theta_i}$. \\
Since we have shown that $\boldsymbol{\hat{f}}$ is assymptotically normal we can use this to derive the asymptotic properties of sample measures like the IOV. The respective Jacobian is given by:
\begin{align*}
\boldsymbol{j}_{\mathrm{IOV}} &= \left(\frac{\partial \mathrm{IOV}(\boldsymbol{f})}{\partial f_0 }, \ldots, \frac{\partial \mathrm{IOV}(\boldsymbol{f})}{\partial f_{m-1} }\right) = \frac{4}{m}\left((1- 2 f_0), \ldots,(1-2f_{m-1}) \right)
\end{align*}
which leads to an asymptotic variance of the IOV:
\begin{align*}
    \sqrt{n}\mathrm{Var}(\widehat{IOV(\boldsymbol{f})}) &\overset{asy.}{=} \boldsymbol{j}_{\mathrm{IOV}} \boldsymbol{\Sigma}_Z \boldsymbol{j}_{\mathrm{IOV}}^{\top} = \sum_{i=0}^{m-1} \sum_{j=0}^{m-1} \frac{4(1- 2 f_i)}{m} \frac{4(1- 2 f_j)}{m} \sigma_{\boldsymbol{Z}; i,j}\\
    &= \frac{16}{m^2} \sum_{i,j = 0}^{m-1} (1-2f_i)(1- 2 f_j) \sigma_{\boldsymbol{Z}; i,j}
\end{align*}
However, since $IOV$ is not a linear function, the first-order delta method only provides information about the asymptotic distribution and variance, not on the finite-sample bias:
\[(\widehat{\mathrm{IOV}} \overset{p}{\rightarrow} \mathrm{IOV}(\boldsymbol{f})) \not\Rightarrow (\mathrm{E}[\widehat{\mathrm{IOV}}] = \mathrm{IOV}(\boldsymbol{f})).\]
For nonlinear transformations, the curvature induces a bias of order $n^{-1}$. Since
$\mathrm{IOV}(\boldsymbol{f}) = \frac{4}{m}\sum_{i=0}^{m-1}f_i(1-f_i)$ is quadratic in $\boldsymbol{f}$, all derivatives of order three and higher vanish
identically. Consequently, the second-order Taylor expansion is not merely an
approximation but exact:
\[
\mathrm{IOV}(\hat{\boldsymbol{f}}) = \mathrm{IOV}(\boldsymbol{f}) 
+ \boldsymbol{j}_{\mathrm{IOV}}(\hat{\boldsymbol{f}} - \boldsymbol{f}) 
+ \frac{1}{2}(\hat{\boldsymbol{f}}-\boldsymbol{f})^\top 
\boldsymbol{H}_{\mathrm{IOV}}(\boldsymbol{f})(\hat{\boldsymbol{f}}-\boldsymbol{f}),
\]
with no remainder term. $\boldsymbol{H}_{\mathrm{IOV}}(\boldsymbol{f})$
is the Hessian matrix such that\[\boldsymbol{H}_{\mathrm{IOV}}(\boldsymbol{f}) = \left(\frac{\partial^2 \mathrm{IOV}(\boldsymbol{f})}{\partial f_i f_j}\right)_{i,j = 0, \ldots, m-1} = \begin{cases}
    -\frac{8}{m} & \text{if } i = j\\
    0 & \text{else}
\end{cases} = -\frac{8}{m} \boldsymbol{I}_m.\]
Since
\[\mathrm{E}(\boldsymbol{j}_{\mathrm{IOV}} (\hat{\boldsymbol{f}} - \boldsymbol{f})) = 0\]
the finite sample bias is given by
\begin{align*}
    \mathrm{Bias}(\mathrm{IOV}(\hat{\boldsymbol{f}})) 
    &= \mathrm{E}\!\left[\frac{1}{2}(\hat{\boldsymbol{f}}- \boldsymbol{f})^\top 
    \boldsymbol{H}_{\mathrm{IOV}}(\boldsymbol{f})(\hat{\boldsymbol{f}}- \boldsymbol{f})\right] \\
    &= \frac{1}{2}\,\mathrm{tr}\!\left(\boldsymbol{H}_{\mathrm{IOV}}\,
    \mathrm{Cov}(\hat{\boldsymbol{f}})\right) 
    \tag*{$\mathrm{E}[\boldsymbol{u}^\top \boldsymbol{A}\boldsymbol{u}] 
    = \mathrm{tr}(\boldsymbol{A}\,\mathrm{Cov}(\boldsymbol{u}))$}\\
    &= \frac{1}{2}\,\mathrm{tr}\!\left(-\frac{8}{m}\boldsymbol{I}_m \cdot 
    \frac{1}{n}\boldsymbol{\Sigma}_Z\right) \\
    &= -\frac{4}{mn} \sum_{i=0}^{m-1} \sigma_{\boldsymbol{Z};\,i,i}.
\end{align*}
This implies that the bias is strictly negative and only depends on the marginal variances $\sigma_{\boldsymbol{Z}; i,i}$ of the indivdual estimators $f_i$,  not on the covariances. The results are summarized in the following theorem: \\
\begin{theorem}[Asymptotic Distribution of $\widehat{\mathrm{IOV}}$]\label{thm:iov_asymp}
Under the assumptions of Theorem~\ref{thm:clt_fhat}:
\begin{enumerate}
    \item[\normalfont(i)] \textbf{Asymptotic variance:}
    \begin{equation}\label{eq:var_iov}
        \sqrt{n}\mathrm{Var}(\widehat{\mathrm{IOV}}) \overset{asy.}{=}
        \frac{16}{m^2}\sum_{i,j=0}^{m-1}(1-2f_i)(1-2f_j)\,\sigma_{Z;i,j}.
    \end{equation}
    \item[\normalfont(ii)] \textbf{Bias to order $n^{-1}$:}
    \begin{equation}\label{eq:bias_iov}
        \mathrm{Bias}(\widehat{\mathrm{IOV}})
        = -\frac{4}{m\,n}\sum_{i=0}^{m-1}\sigma_{Z;i,i}.
    \end{equation}
\end{enumerate}
\end{theorem}
\subsection*{Skewness}
In a similiar way the asymptotic properties of other dispersion metrics such as the skew can be analyzed. A simple measure for the skew of an ordinal process is given by:
\[\mathrm{skew}(\boldsymbol{f}) = \frac{2}{m} \left(\sum_{i=0}^{m-1}f_i\right) - 1.\]
The measure captures asymmetry through the average level of the CDF: A symmetric
distribution is taken as baseline and deviations 
from this baseline reflect asymmetry in the distribution. The measure is scaled to
$[-1, 1]$, and the three extreme cases illustrate its behavior clearly:
\begin{itemize}
    \item \textbf{Left-skewed} ($\mathrm{skew} = 1$): all probability mass is 
    concentrated in the lowest category $s_0$, so $f_i = 1$ for all $i$ and
    $\mathrm{skew} = \frac{2}{m} \cdot m - 1 = 1$.
    \item \textbf{Symmetric} ($\mathrm{skew} = 0$): the distribution is symmetric
    around the midpoint, which implies $f_i+ f_{m-1-i} = 1$ for all $i$. It follows that $\sum_{i=0}^{m-1} f_i = m/2$ and
    $\mathrm{skew} = \frac{2}{m} \cdot \frac{m}{2} - 1 = 0$.
    \item \textbf{Right-skewed} ($\mathrm{skew} = -1$): all probability mass is
    concentrated in the highest category $s_m$, so $f_i = 0$ for all $i < m$ and
    $\mathrm{skew} = \frac{2}{m} \cdot 0 - 1 = -1$.
\end{itemize}
The jacobian for the skew is given by
\[\boldsymbol{j}_{skew} = \left(\frac{\partial ~ \mathrm{skew}(\boldsymbol{f})}{\partial f_0}, \ldots, \frac{\partial ~\mathrm{skew}(\boldsymbol{f})}{\partial f_{m-1}}\right) = \left(\frac{2}{m}, \ldots, \frac{2}{m}\right)\]
such that we end up with the following results:
\begin{proposition}[Asymptotic Properties of $\widehat{\mathrm{skew}}$]\label{prop:skew}
Under the assumptions of Theorem~\ref{thm:clt_fhat}:
\begin{equation}\label{eq:var_skew}
    \sqrt{n}\mathrm{Var}(\widehat{\mathrm{skew}}) \overset{asy.}{=}
    \frac{4}{m^2}\sum_{i,j=0}^{m-1}\sigma_{Z;ij}.
\end{equation}
Since $\mathrm{skew}$ is linear in $\boldsymbol{f}$, the Hessian vanishes and the
estimator is exactly unbiased for all $n$.
\end{proposition}
\chapter{Introducing Missingness}
\section{Missingness Mechanisms}

In many applied settings, categorical time series are not fully 
observed. Individual time points may be missing due to sensor 
failures, non-response in survey data, or dropout in longitudinal 
studies. Before modelling the statistical consequences of 
missingness, it is useful to distinguish between different 
mechanisms that may give rise to it, as these have fundamentally 
different implications for estimation.

The classical taxonomy of missing data mechanisms goes back to 
\citet{Rubin1976}, who distinguished between three types based on 
the dependence structure between the missingness and the underlying 
data. While Rubin's framework was originally developed for 
multivariate cross-sectional data, the core concepts translate 
naturally to the time series setting when conditioning is extended 
to the full history of the process. Denoting by $O_t \in \{0,1\}$ 
the indicator of whether $X_t$ is observed (to be formally 
introduced in Section~\ref{sec:Amp Mod}), and by 
$\mathcal{F}_{t-1} := \sigma(X_s, O_s : s < t)$ the history up 
to time $t$, the three mechanisms are defined as follows.

\begin{definition}[Missingness Mechanisms]
\label{def:missingness}
\begin{enumerate}[(i)]
    \item \textbf{Missing Completely at Random (MCAR):} The 
    observation probability does not depend on the current or 
    past values of the process:
    \[P(O_t = 1 \mid X_t,\, \mathcal{F}_{t-1}) = \pi.\]

    \item \textbf{Missing at Random (MAR):} The observation 
    probability depends on the observed history, but not on the 
    current (possibly unobserved) value:
    \[P(O_t = 1 \mid X_t,\, \mathcal{F}_{t-1}) 
    = P(O_t = 1 \mid \mathcal{F}_{t-1}).\]

    \item \textbf{Missing Not at Random (MNAR):} The observation 
    probability depends on the current value $X_t$ itself, which 
    may be unobserved:
    \[P(O_t = 1 \mid X_t,\, \mathcal{F}_{t-1}) 
    = P(O_t = 1 \mid X_t,\, \mathcal{F}_{t-1}).\]
\end{enumerate}
\end{definition}

\begin{remark}
    While Rubin's taxonomy was formulated for i.i.d.\ data, the 
    adaptation to the time series setting is conceptually 
    straightforward: the key distinction remains whether missingness 
    depends on the unobserved current value $X_t$ (MNAR) or only 
    on the observable past (MAR/MCAR). To the best of our 
    knowledge, a formal systematic treatment of this adaptation 
    for univariate categorical time series has not appeared in 
    the literature.
\end{remark}

The assumption under which the subsequent results are derived is 
that $(O_t)$ and $(X_t)$ are \emph{independent}, which corresponds 
to the MCAR case. This is the strongest of the three assumptions 
and implies in particular that the corrected estimator $\hat{f}^*$ 
introduced in Section~\ref{sec:Amp Mod} is consistent. Under MAR 
or MNAR, the independence assumption fails and the estimator may 
be biased — a point we return to in Section~\ref{sec:mnar}.

\section{Amplitude Modulation}\label{sec:Amp Mod}
\subsection*{Definition of the observation process}
To model the missingness, the concept of \textbf{amplitude modulation} is employed. This approach models missingness through a binary stochastic process $(O_t)_t$ that runs concurrently with the main process $(X_t)_t$ and modulates its "amplitude"  in a binary manner:

\[
O_t \in \{0,1\} \quad \forall t
\]
where $O_t = 1$ indicates that $X_t$ is fully observed (amplitude = 1), and $O_t = 0$ indicates that $X_t$ is missing (amplitude = 0).

The amplitude modulating process $O_t$ can be:
\begin{itemize}
    \item \textbf{Deterministic}: For each $t$, $O_t = o_t$ where $o_t \in \{0,1\}$ is a known, fixed value.
    \item \textbf{Stochastic}: $O_t$ follows a probability distribution with $\pi_t := P(O_t=1)$, possibly dependent on the history of $X_t$, $O_t$, or external factors.
\end{itemize}
\subsection*{Bias of the naive estimator}
The amplitude modulation is applied to the binarization vector $\boldsymbol{Z}_t$ to obtain the observable process $(O_t \boldsymbol{Z}_t)$. We then define the naive estimator of $\boldsymbol{f}$ as:
\[\hat{\boldsymbol{\mathfrak{f}}}:= \frac{1}{n} \sum_{t=1}^{n}O_t \boldsymbol{Z}_t.\]
To assess the performance of this naive estimator we compute its expectation
\[\mathrm{E}[\hat{\boldsymbol{\mathfrak{f}}}] = \frac{1}{n} \sum_{t = 1}^{n} \mathrm{E}[O_t\boldsymbol{Z_t}] = \frac{1}{n} \sum_{t = 1}^{n} \mathrm{E}[\boldsymbol{Z_t} \mathrm{E}[O_t|\boldsymbol{Z}_t]],\]
where the last equality follows from the law of iterated expectations.\\
The dependendce structure between $(O_t)_t$ and $(\boldsymbol{Z}_t)_t$ (or equivalently $(X_t)_t$) is crucial. Under the assumption of independence, the expression simplifies to
\[\frac{1}{n} \sum_{t = 1}^{n} \mathrm{E}[\boldsymbol{Z_t} \mathrm{E}[O_t|\boldsymbol{Z}_t]] = \frac{1}{n}\sum_{t=1}^{n} \mathrm{E}[\boldsymbol{Z}_t]  \mathrm{E}[O_t] = \left(\frac{1}{n} \sum_{t=1}^{n} \mathrm{E}[O_t]\right) \boldsymbol{f} \]
\subsection*{The missingness corrected estimator}
\begin{definition}[Corrected Estimator of $\boldsymbol{f}$]\label{def:correctedest}
Since $\mathrm{E}[\hat{\boldsymbol{\mathfrak{f}}}]$ deviates from $\boldsymbol{f}$
by the factor $\frac{1}{n}\sum_{t=1}^{n}\mathrm{E}[O_t]$, the corrected estimator
of $\boldsymbol{f}$ is defined as
\[
    \hat{\boldsymbol{f}}^*
    := \frac{\frac{1}{n} \sum_{t=1}^{n}O_t \boldsymbol{Z}_t}
            {\frac{1}{n}\sum_{t=1}^{n} O_t}
    =: \frac{\hat{\boldsymbol{\mathfrak{f}}}}{\bar{O}}.
\]
\end{definition}
Under the assumption of the independence of $(O_t)$ and $(X_t)$, the stochastic properties of the corrected estimator $\hat{\boldsymbol{f^*}}$ can be studied. If $(O_t)$ is deterministic, then $\mathrm{E}[O_t] = o_t \in \{0,1\}$ for any given $t$. It follows that
\[\frac{1}{n}\sum_{t=1}^{n} \mathrm{E}[O_t] = \frac{1}{n}\sum_{t=1}^{n} O_t = \bar{O},\]
and therefore, here $\hat{\boldsymbol{f}^*} = \hat{\boldsymbol{\mathfrak{f}}}/\bar{O}$ is a perfectly unbiased estimator of $\boldsymbol{f}$. However, if $O_t$ is a stochastic process then $\mathrm{E}[O_t] = \pi_t \in (0; 1]$ and
\[\frac{1}{n}\sum_{t=1}^{n} \mathrm{E}[O_t]= \frac{1}{n}\sum_{t=1}^{n} \pi_t \neq \frac{1}{n}\sum_{t=1}^{n} O_t = \bar{O},\]
to where $\hat{\boldsymbol{f^*}}$ is generally biased.
\section{Asymptotic Theory for the corrected estimator}
\subsection*{The extended process - stationarity and $\alpha$-mixing}
To study this bias we assume $(O_t)$ and $(X_t)$ to be stationary and $\alpha$-mixing with exponetially decaying coefficients, with $\mathrm{E}[O_t] = \pi $ and a autocovariance function $\gamma_{O}(h) := \mathrm{Cov}[O_h, O_0]$ and joint expectations $\pi(h):= \mathrm{E}[O_h O_0] = \gamma_O(h) + \pi^2$. To study the joint asymptotic distribution of the estimator $\hat{\boldsymbol{f}}^*$, we introduce the extended binary vectors in order to derive the joint asymptotic distribution of all sample averages entering its definition:
\[\boldsymbol{Z}^*_t:= (O_t, O_t Z_{t,0}, \ldots, O_t Z_{t,m-1})^\top =  O_t (1, \boldsymbol{Z}^\top_t)^\top,\]
where the elements of $\boldsymbol{Z}_t^*$ are are numbered as $Z_{t,-1}^*, Z_{t,0}^*, \ldots, Z_{t,m-1}^*$The expectation of $\boldsymbol{Z}_t^*$ is given by
\begin{align*}
    \mathrm{E}[\boldsymbol{Z}^{*}_t] &= (\mathrm{E}[O_t], \mathrm{E}[O_t Z_{t,0}], \ldots, \mathrm{E}[O_t Z_{t,m-1}])^\top \\
    &= (\mathrm{E}[O_t], \mathrm{E}[O_t] \mathrm{E}[Z_{t,0}], \ldots, \mathrm{E}[O_t] \mathrm{E}[Z_{t,m-1}])^\top \tag{Independence}\\
    &= (\pi, \pi f_0, \ldots, \pi f_{m-1})^\top \\
    &=\pi (1, \boldsymbol{f}^{\top})^\top =: \boldsymbol{\mu}_{\boldsymbol{Z}_t^{*}}.
\end{align*}
Since the processes $(O_t)$ and $(\boldsymbol{Z}_t)$ are independent and stationary, the joint process $((O_t, \boldsymbol{Z}_t)^{\top})$ is also stationary. Moreover, if both processes are $\alpha$-mixing, then the joint process is $\alpha$-mixing. As $\boldsymbol{Z}_t^*$ is a measurable function of $(O_t, \boldsymbol{Z})^\top$ it follows that $\boldsymbol{Z}^*_t$ is stationary and $\alpha$-mixing (with exponentially decaying coefficients). \\
\subsection*{Multivariate CLT}
To study the asymptotic properties of $\boldsymbol{Z}^*_t$ we once again use Cramer-Wold. Let
\[U_t^* = \boldsymbol{a}^\top\boldsymbol{Z}^*_t, \quad \boldsymbol{a}\in \mathbb{R}^{m+1}.\]
The argument for why $U_t^*$ satisfies the conditions of Ibragimovs CLT isthe same as in the argument for why the linear projections of the non-missing estimator $\hat{\boldsymbol{f}}_t$ converge: Since $U_t^* = \boldsymbol{a}^\top \boldsymbol{Z}_t^*$ is a measurable function of the strictly stationary and $\alpha$-mixing process $\boldsymbol{Z}_t^*$, it inherits both properties, with $\alpha_{U_t^*}(n) \leq \alpha_{\boldsymbol{Z}_t^*}(n) \leq C\rho^n$ following from the inclusion $\sigma(U_t^*) \subset \sigma(\boldsymbol{Z}_t^*)$. The summability condition therefore holds by the same geometric series argument as before. Finally, since each $Z_{t,i}^* \in \{0,1\}$, the projection $U_t^*$ is bounded and all moments are finite. It follows that
\[\sqrt{n}(\bar{U}^*_t- \mu_{U^*_t}) \xrightarrow[n \rightarrow \infty]{d} \mathcal{N}(0, \sigma_{a}^*)\]
where $\mu_{U_t^*} = \mathrm{E}(U_t) = \boldsymbol{a}^\top \boldsymbol{\mu}_{\boldsymbol{Z}_t^*}$ and
\begin{align*}
    \sigma^*_{a} &= \mathrm{Var}(U_t^*) + 2\sum_{h=1}^{\infty}\mathrm{Cov}(U_t, U_{t+h}) \\
    \text{where}\quad \mathrm{Var}(U_t^*) &= \boldsymbol{a}^\top \mathrm{Var}(\boldsymbol{Z}^*_t) \boldsymbol{a} = \sum_{i, j = - 1}^{m-1} a_{i+1} a_{j+1} \mathrm{Cov}(Z^*_{t, i}, Z^*_{t, j})\\
    \text{and} \quad \mathrm{Cov}(U_t^*, U_{t+h}^*) &= \boldsymbol{a}^\top \mathrm{Cov}(\boldsymbol{Z}^*_t, \boldsymbol{Z}^{*\top}_{t+h}) \boldsymbol{a} \\
    \text{such that } \quad \sigma^*_{a} &= \boldsymbol{a}^\top (\mathrm{Var}(\boldsymbol{Z}^*_t) + 2 \sum_{h = 1 }^{\infty} \mathrm{Cov}(\boldsymbol{Z}^*_t, \boldsymbol{Z}^{*}_{t+h})) \boldsymbol{a}.
\end{align*}
Since every linear projection $U_t^*$ is asymptotically normal we can follow that the sample mean
\[\bar{\boldsymbol{Z}}^*_n := \frac{1}{n} \sum_{t=1}^{n} \boldsymbol{Z}^*_t\]
satisfies the multivariat CLT
\[\sqrt{n} \left( \bar{\boldsymbol{Z}}^*_n - \boldsymbol{\mu}_{\boldsymbol{Z}_t^*} \right) \xrightarrow[n \to \infty]{d} \mathcal{N}\left( \boldsymbol{0}, \boldsymbol{\Sigma}_{\boldsymbol{Z}^*} \right)\]
where $\boldsymbol{\Sigma}_{\boldsymbol{Z}^*}$ must sastisfy for all $a \in \mathbb{R}^{m+1}$
\[\boldsymbol{a}^\top\boldsymbol{\Sigma}_{\boldsymbol{Z}^*} \boldsymbol{a} = \sigma^*_a = \boldsymbol{a}^\top (\mathrm{Var}(\boldsymbol{Z}^*_t) + 2 \sum_{h = 1 }^{\infty} \mathrm{Cov}(\boldsymbol{Z}^*_t, \boldsymbol{Z}^{*}_{t+h})) \boldsymbol{a}\]
which indicates
\[\boldsymbol{\Sigma}_{\boldsymbol{Z}^*} = \mathrm{Var}(\boldsymbol{Z}^*_t) + 2\sum_{h=1}^{\infty}\mathrm{Cov}(\boldsymbol{Z}_{t}^*, \boldsymbol{Z}_{t+h}^*).\]
The individudal elements of $\boldsymbol{\Sigma}_{\boldsymbol{Z}^*}$ are given by
{\small
\begin{align*}
&\textbf{(i)}\ i = j = -1: \\
&\quad \mathrm{Var}(O_t) + 2\sum_{h=1}^\infty \mathrm{Cov}(O_t, O_{t+h}) 
= \pi(1-\pi) + 2\sum_{h=1}^{\infty}\gamma_O(h) \\[2ex]
&\textbf{(ii)}\ j > i = -1: \\
&\quad \mathrm{Cov}(O_t, O_tZ_{t,j}) 
+ 2\sum_{h=1}^\infty \mathrm{Cov}(O_t, O_{t+h}Z_{t+h,j}) \\
&\quad \overset{\perp}{=} 
\left(\mathrm{E}[O_t^2]\mathrm{E}[Z_{t,j}] - \mathrm{E}[O_t]^2\mathrm{E}[Z_{t,j}]\right) \\
&\qquad + 2\sum_{h=1}^{\infty}
\left(\mathrm{E}[O_tO_{t+h}]\mathrm{E}[Z_{t,j}] - \mathrm{E}[O_t]^2\mathrm{E}[Z_{t,j}]\right) \\
&\quad \overset{O_t \in \{0,1\}}{=} 
(\pi - \pi^2)f_j + 2\sum_{h=1}^{\infty}(\gamma_O(h) + \pi^2 - \pi^2)f_j \\
&\quad = \pi(1-\pi)f_j + 2f_j\sum_{h=1}^{\infty}\gamma_O(h) 
= f_j\!\left(\pi(1-\pi) + 2\sum_{h=1}^{\infty}\gamma_O(h)\right) 
= f_j\,\sigma_{\boldsymbol{Z}^*;\,-1,-1} \\[2ex]
&\textbf{(iii)}\ i,j \geq 0: \\
&\quad \mathrm{Cov}(O_tZ_{t,i}, O_tZ_{t,j}) 
+ \sum_{h=1}^{\infty}\bigl[\mathrm{Cov}(O_tZ_{t,i}, O_{t+h}Z_{t+h,j}) 
+ \mathrm{Cov}(O_{t+h}Z_{t+h,i}, O_tZ_{t,j})\bigr] \\
&\quad \overset{\perp}{=} 
\bigl(\mathrm{E}[O_t^2]\mathrm{E}[Z_{t,i}Z_{t,j}] - \pi^2f_if_j\bigr) \\
&\qquad + \sum_{h=1}^{\infty}\bigl[
\mathrm{E}[O_tO_{t+h}]\mathrm{E}[Z_{t,i}Z_{t+h,j}] - \pi^2f_if_j 
+ \mathrm{E}[O_tO_{t+h}]\mathrm{E}[Z_{t,j}Z_{t+h,i}] - \pi^2f_if_j
\bigr]\\
&\quad = \pi f_{\mathrm{min} \{i,j\}} - \pi^2 f_if_j  + \sum_{h=1}^{\infty} \pi(h) f_{ij}(h) + \pi(h) f_{ji}(h) - 2\pi^2 f_if_j\\
&\quad = \pi f_{\mathrm{min} \{i,j\}} - \pi^2 f_if_j  + \pi f_i f_j - \pi f_i f_j  \\
&\qquad + \sum_{h=1}^{\infty} \pi(h) f_{ij}(h) + \pi(h) f_{ji}(h) - 2\pi^2 f_if_j -  2\gamma(h) f_if_j +  2\gamma(h) f_if_j \\
&\quad = \pi (f_{\mathrm{min} \{i,j\}}- f_if_j) + \left(\pi(1-\pi)   + 2 \sum_{h=1}^{\infty}  \gamma(h)\right) f_if_j + \sum_{h=1}^{\infty} \pi(h) (f_{ij}(h) + f_{ji}(h) - 2f_if_j) \\
&\quad = f_if_j \sigma_{\boldsymbol{Z}^*;-1, -1} +\pi (f_{\mathrm{min} \{i,j\}}- f_if_j)  + \sum_{h=1}^{\infty} \pi(h) (f_{ij}(h) + f_{ji}(h) - 2f_if_j)
\end{align*}
}
To now derive the asymptotic properties of $\hat{\boldsymbol{f}}^*$ we define the vector-valued function $\boldsymbol{g}:= [0,1]^{m+1} \rightarrow [0,1]^{m}$ as
\[g_j(x_{-1}, x_{0}, \ldots, x_{m-1}) = x_j / x_{-1}, \quad \text{for } j = 0 \ldots, m-1,\]
such that
\begin{align*}
    \boldsymbol{g}(\bar{\boldsymbol{Z}^*_t}) &= (g_0(\bar{\boldsymbol{Z}}^*_t), \ldots, g_{m-1}(\bar{\boldsymbol{Z}}^*_t))^\top  =  (\hat{f}_0 O_t/O_t, \ldots, \hat{f}_{m-1} O_t/O_t)^\top\\
    &= \hat{\boldsymbol{f}}^*
\end{align*}
and  accordingly
\begin{align*}
    g(\bar{\boldsymbol{\mu}}_{\boldsymbol{Z}^*}^*) = \boldsymbol{f}.
\end{align*}
The Jacobian matrix of $\boldsymbol{g}(\boldsymbol{x})$ evaluated at $\boldsymbol{x} = \boldsymbol{\mu}_{\boldsymbol{Z}^*}$ is given by
\[\boldsymbol{G}_{(m+1)\times m} =
\begin{pmatrix}
    \frac{\partial g_0(\boldsymbol{\mu}_{\boldsymbol{Z}^*})}{\partial x_{-1}}& \ldots & \frac{\partial g_0(\boldsymbol{\mu}_{\boldsymbol{Z}^*})}{\partial x_{m-1}}\\
    \vdots & \cdots& \vdots\\
    \frac{\partial g_{m-1}(\boldsymbol{\boldsymbol{\mu}_{\boldsymbol{Z}^*}})}{\partial x_{-1}}& \ldots & \frac{\partial g_{m-1}(\boldsymbol{\mu}_{\boldsymbol{Z}^*})}{\partial x_{m-1}}
\end{pmatrix} =
\begin{pmatrix}
    -\frac{f_0}{\pi} &  \frac{1}{\pi} & 0 & \ldots & 0\\
    -\frac{f_1}{\pi} & 0 & \frac{1}{\pi}&\ddots & \vdots \\
    \vdots & \vdots & \ddots & \ddots & 0\\
    -\frac{f_{m-1}}{\pi}& 0 &\ldots& 0 & \frac{1}{\pi}
\end{pmatrix}\]
Then, following the deltamethod, we have the following result
\begin{theorem}[Asymptotic Normality of $\hat{\boldsymbol{f}}^*$]\label{thm:clt_fstarhat}
Let $(X_t)$ be a strictly stationary, ordinal process that is $\alpha$-mixing with exponentially decaying
coefficients. Let the modulating process $(O_t)$ also be strictily stationary and $\alpha$-mixing with exponentially decaying coefficients, with moments $\mathrm{E}[O_t] = \pi > 0$ and $\mathrm{E}[O_h O_0] = \pi(h) > 0$. Furthermore, let $(O_t)$ and $(X_t)$ be independent, where $X_t$ is observed if $O_t = 1$. Then the corrected estimator from definition~\ref{def:correctedest} satisfies the following multivariate CLT:
\begin{equation}\label{eq:clt_fstarhat}
    \sqrt{n}\bigl(\hat{\boldsymbol{f}}^* - \boldsymbol{f}\bigr)
    \xrightarrow[n\to\infty]{d}
    \mathcal{N}\bigl(\boldsymbol{0}, \boldsymbol{\Sigma}_{\hat{\boldsymbol{f}}^*}\bigr),
    \quad \text{with } \boldsymbol{\Sigma}_{\hat{\boldsymbol{f}}^*} = (\sigma_{\hat{\boldsymbol{f}}^*;i,j})_{i,j=0,\ldots,m-1},
\end{equation}
where
\begin{equation}\label{eq:sigma_fstar}
    \sigma_{\hat{\boldsymbol{f}}^*;i,j} = \frac{1}{\pi} (f_{\min \{i,j\}}- f_if_j) + \frac{1}{\pi^2} \sum_{h=1}^{\infty}\pi(h)(f_{ij}(h)+f_{ji}(h)-2f_if_j).
\end{equation}\end{theorem}
The structure of $\boldsymbol{\Sigma}_{\hat{\boldsymbol{f}^*}}$ can be computed as follows
\begin{align*}
    \boldsymbol{\Sigma}_{\hat{\boldsymbol{f}^*}} &= \boldsymbol{G} \boldsymbol{\Sigma}_{\boldsymbol{Z}^*} \boldsymbol{G}^\top \\
    &= \begin{pmatrix}
    -\frac{f_0}{\pi} &  \frac{1}{\pi} & 0 & \ldots & 0\\
    -\frac{f_1}{\pi} & 0 & \frac{1}{\pi}&\ddots & \vdots \\
    \vdots & \vdots & \ddots & \ddots & 0\\
    -\frac{f_{m-1}}{\pi}& 0 &\ldots& 0 & \frac{1}{\pi}
    \end{pmatrix}
    \begin{pmatrix}
        \sigma_{\boldsymbol{Z}^*; -1, -1}& \ldots & \sigma_{\boldsymbol{Z}^*; -1, m-1}\\
        \vdots & \ddots &\vdots \\
        \sigma_{\boldsymbol{Z}^*; m-1, -1} & \ldots & \sigma_{\boldsymbol{Z}^*; m-1, m-1}
    \end{pmatrix}
    \boldsymbol{G}^\top\\
    &= \begin{pmatrix}
        -\frac{f_0}{\pi}\sigma_{\boldsymbol{Z}^*; -1,-1} + \frac{1}{\pi}\sigma_{\boldsymbol{Z}^*; 0, -1} &\ldots&-\frac{f_0}{\pi}\sigma_{\boldsymbol{Z}^*; -1, m-1} + \frac{1}{\pi}\sigma_{\boldsymbol{Z}^*; 0, m-1} \\
        \vdots & \ddots &\vdots \\
        -\frac{f_{m-1}}{\pi}\sigma_{\boldsymbol{Z}^*; -1, -1} + \frac{1}{\pi}\sigma_{\boldsymbol{Z}^*; m-1, -1} &\ldots&-\frac{f_{m-1}}{\pi}\sigma_{\boldsymbol{Z}^*; -1, m-1} + \frac{1}{\pi}\sigma_{\boldsymbol{Z}^*; m-1, m-1} \\
    \end{pmatrix}\boldsymbol{G}^\top.
\end{align*}
Such that the indivual lements look like this:
\begin{align*}
    \sigma_{\hat{\boldsymbol{f}}^*; i, j}  &= (-\frac{f_j}{\pi}) (-\frac{f_i}{\pi}\sigma_{\boldsymbol{Z}^*; -1, -1} + \frac{1}{\pi} \sigma_{\boldsymbol{Z}^*; i, -1}) + \frac{1}{\pi}(-\frac{f_{i}}{\pi}\sigma_{\boldsymbol{Z}^*; -1, j}+ \frac{1}{\pi} \sigma_{\boldsymbol{Z}^*; i, j}) \\
    &= \frac{f_j f_i }{\pi^2} \sigma_{\boldsymbol{Z}^*; -1, -1} - \frac{f_j}{\pi^2} \sigma_{\boldsymbol{Z}^*; i, -1} - \frac{f_i}{\pi^2}\sigma_{\boldsymbol{Z}^*; -1, j} + \frac{1}{\pi^2} \sigma_{\boldsymbol{Z}^*; i, j} \\
    &= \frac{1}{\pi^2}\left(f_if_j\sigma_{\boldsymbol{Z}^*; -1, -1} - f_if_j\sigma_{\boldsymbol{Z}^*; -1, -1} - f_if_j\sigma_{\boldsymbol{Z}^*; -1, -1} + \sigma_{\boldsymbol{Z}^*; i, j}\right) = \frac{1}{\pi^2}(\sigma_{\boldsymbol{Z}^*; i, j}-f_if_j \sigma_{\boldsymbol{Z}^*; -1, -1}) \\
    &= \frac{1}{\pi^2}\left(\left[f_if_j\sigma_{\boldsymbol{Z}^*; -1, -1} + \pi(f_{\min \{i,j\}}- f_if_j) + \sum_{h=1}^{\infty}\pi(h)f_{ij}(h)+f_{ji}(h)-2f_if_j\right]- f_if_j \sigma_{\boldsymbol{Z}^*; -1, -1}\right) \\
    &= \underbrace{\frac{1}{\pi} (f_{\min \{i,j\}}- f_if_j) }_{\mathrm{lag}(0)}+ \underbrace{\sum_{h=1}^{\infty}\pi(h)(f_{ij}(h)+f_{ji}(h)-2f_if_j)}_{\mathrm{lag}(h)}
\end{align*}
Here, the lag-0 term captures the contemporaneous covariance structure of the
observed process, while the lag-$h$ terms accumulate the serial dependence
across all lags $h \geq 1$.
\subsection*{Bias analysis}
Since $g_j$ is not a linear function of $\boldsymbol{Z}_t^*$, the bias of order
$O(n^{-1})$ is computable using a second-order Taylor expansion. The Hessian of $g_j$ evaluated in $\boldsymbol{\mu}_{\boldsymbol{Z}^*}$ is given by
\[
\boldsymbol{H}_{g_j} = \left(\frac{\partial^2 g_j(\mu_{\boldsymbol{Z}^*})}{\partial x_i \partial x_k}\right)_{i,k=-1,\ldots,m-1} =
\begin{cases}
    \frac{2f_j}{\pi^2}  & \text{if } i = k = -1 \\
    -\frac{1}{\pi^2}    & \text{if } i = -1,\ k = j \\
    0                   & \text{else.}
\end{cases}
\]
Since only two entries of $\boldsymbol{H}_{g_j}$ are nonzero, and using the symmetry
$\sigma_{\boldsymbol{Z}^*;-1,j} = \sigma_{\boldsymbol{Z}^*;j,-1}$, the bias reduces to
\begin{align*}
    \mathrm{Bias}(\hat{f}_j)
    &= \frac{1}{2}\mathrm{tr}(\boldsymbol{H}_{g_j}\boldsymbol{\Sigma}_{\boldsymbol{Z}^*}) \\
    &= \frac{1}{2}h_{g_j;-1,-1}\,\sigma_{\boldsymbol{Z}^*;-1,-1}
        + h_{g_j;-1,j}\,\sigma_{\boldsymbol{Z}^*;-1,j} \\
    &= \frac{1}{2}\cdot\frac{2f_j}{\pi^2}\,\sigma_{\boldsymbol{Z}^*;-1,-1}
        - \frac{1}{\pi^2}\cdot f_j\,\sigma_{\boldsymbol{Z}^*;-1,-1} \\
    &= \frac{f_j}{\pi^2}\,\sigma_{\boldsymbol{Z}^*;-1,-1}
        - \frac{f_j}{\pi^2}\,\sigma_{\boldsymbol{Z}^*;-1,-1} = 0,
\end{align*}
where the last step uses $\sigma_{\boldsymbol{Z}^*;-1,j} = f_j\,\sigma_{\boldsymbol{Z}^*;-1,-1}$
from case (ii) of $\boldsymbol{\Sigma}_{\boldsymbol{Z}^*}$.\\
\subsection*{Special Cases}
To further study the asymptotic distribution of $\hat{\boldsymbol{f}}^*$,
we consider three special cases of the general result.

\begin{corollary}[Full Observation]
If the process is fully observed, i.e.\ $\pi = 1$, then
\[
    \sigma_{\hat{\boldsymbol{f}}^*;\, i,j}
    = \bigl(f_{\min\{i,j\}} - f_if_j\bigr)
    + \sum_{h=1}^{\infty} f_{ij}(h) + f_{ji}(h) - 2f_if_j
    = \sigma_{\boldsymbol{Z};\, i,j}.
\]
\end{corollary}
That is, the asymptotic covariance of $\hat{\boldsymbol{f}}^*$ reduces to that of
the complete-data estimator $\hat{\boldsymbol{f}}$(\ref{thm:clt_fhat}), as one would expect.

\begin{corollary}[i.i.d.\ Observation Process]
If $(O_t)$ is i.i.d., then $\pi(h) = \pi^2$ for all $h \geq 1$,
and the asymptotic covariance simplifies to
\[
    \sigma_{\hat{\boldsymbol{f}}^*;\, i,j}
    = \frac{1}{\pi}\bigl(f_{\min\{i,j\}} - f_if_j\bigr)
    + \sum_{h=1}^{\infty} f_{ij}(h) + f_{ji}(h) - 2f_if_j.
\]
\end{corollary}
Compared to the complete-data case, the lag-0 term is inflated by a factor of
$1/\pi \geq 1$, reflecting the additional uncertainty introduced by the random
missingness. The lag-$h$ terms for $h \geq 1$, however, are unaffected, since
i.i.d.\ missingness introduces no additional serial dependence into the observation
process.

\begin{corollary}[i.i.d.\ Main Process]
If $(X_t)$ is i.i.d., then $f_{ij}(h) = f_if_j$ for all
$i,j = 0,\ldots,m-1$ and $h \geq 1$, so that the asymptotic
covariance reduces to
\[
    \sigma_{\hat{\boldsymbol{f}}^*;\, i,j}
    = \frac{1}{\pi}\bigl(f_{\min\{i,j\}} - f_if_j\bigr)
    + \frac{1}{\pi^2}\sum_{h=1}^{\infty} \pi(h)
    \overbrace{(f_{ij}(h) + f_{ji}(h) - 2f_if_j)}^{=0}
    = \frac{1}{\pi}\bigl(f_{\min\{i,j\}} - f_if_j\bigr).
\]
\end{corollary}
When the main process is serially independent, all temporal dependence in
$\hat{\boldsymbol{f}}^*$ vanishes and the asymptotic covariance only depends on the lag-$0$ term. 

\section{Asymptotic marginal properties}
The asymptotic properties of marginal statistics based on $\hat{\boldsymbol{f}}^*$
follow directly from the delta method, in exact analogy to the complete-data case.
The following results are for the IOV and the skew. However, the derivations for other marginal statistics areanalogous- 
\begin{corollary}[Asymptotic distribution of $\mathrm{IOV}(\hat{\boldsymbol{f}}^*)$]
\label{cor:iov_missing}
Under the conditions of Theorem~\ref{thm:clt_fstarhat}, the delta-method argument from Theorem~\ref{thm:iov_asymp} 
carries over to the missing-data case with $\boldsymbol{\Sigma}_Z$ replaced by
$\boldsymbol{\Sigma}_{\hat{\boldsymbol{f}}^*}$, yielding
\[
    \sqrt{n}\mathrm{Var}\bigl(\mathrm{IOV}(\hat{\boldsymbol{f}}^*)\bigr)
    \overset{\mathrm{asy.}}{=}
    \frac{16}{m^2} \sum_{i,j=0}^{m-1}(1-2f_i)(1-2f_j)\,\sigma_{\hat{\boldsymbol{f}}^*;\,i,j}.
\]
Moreover, since $\boldsymbol{H}_{\mathrm{IOV}} = -\frac{8}{m}\boldsymbol{I}_m$ remains
diagonal, the bias to order $n^{-1}$ depends only on the marginal variances of
$\hat{\boldsymbol{f}}^*$:
\[
    \mathrm{Bias}\bigl(\mathrm{IOV}(\hat{\boldsymbol{f}}^*)\bigr)
    = -\frac{4}{mn}\sum_{i=0}^{m-1}\sigma_{\hat{\boldsymbol{f}}^*;\,i,i}.
\]
\end{corollary} 
\begin{remark}[Skew under missingness]
The analogous result for the skew follows immediately, since the skew is linear in
$\boldsymbol{f}$ and hence $\boldsymbol{H}_{\mathrm{skew}} = 0$. In particular, the
estimator is unbiased and its asymptotic variance is given by
\[
    \sqrt{n}\mathrm{Var}\bigl(\mathrm{skew}(\hat{\boldsymbol{f}}^*)\bigr)
    \overset{\mathrm{asy.}}{=}
    \frac{4}{m^2}\sum_{i,j=0}^{m-1}\sigma_{\hat{\boldsymbol{f}}^*;\,i,j}.
\]
\end{remark}

\section{Serial Dependence}
\label{sec:serial-dependence}

So far, we have only examined aspects of the marginal distribution of 
the categorical process $(X_t)$. However, an essential component of 
time series analysis is the study of serial dependence. Recall that 
the most widely used measures to describe such dependence are the 
ordinal and nominal Cohen's $\kappa$ (Definition \ref{def:cohens-kappa}). To estimate these measures, we 
again rely on relative frequencies:
\[
    \hat{\kappa}(h)_{ord} = \frac{\sum_{j=0}^{m-1}\left(\hat{f}_{jj}^*(h) 
    - \bigl(\hat{f}^*_j\bigr)^2\right)}{\sum_{i=0}^{m-1}
    \hat{f}^*_i\bigl(1-\hat{f}_i^*\bigr)}.
\]
Hence, it remains to define the estimator $\hat{f}^*_{ij}(h)$ for the 
bivariate lag-$h$ probabilities $f_{ij}(h)$. As before, we begin with 
the naive estimator given by the bivariate relative frequency of the 
amplitude-modulated binarizations,
\[
    \hat{\mathfrak{f}}_{ij}(h) 
    = \frac{1}{n-h}\sum_{t=h+1}^{n} O_t\, O_{t-h}\, Z_{t,i}\, Z_{t-h,j},
\]
and compute its expectation:
\begin{align*}
    \mathrm{E}\!\left[\hat{\mathfrak{f}}_{ij}(h)\right]
    &= \frac{1}{n-h}\sum_{t=h+1}^{n}
       \mathrm{E}\!\left[O_t O_{t-h} Z_{t,i} Z_{t-h,j}\right] \\
    &= \frac{1}{n-h}\sum_{t=h+1}^{n}
       \mathrm{E}\!\left[O_t O_{t-h}\right] \cdot 
       \mathrm{E}\!\left[Z_{t,i} Z_{t-h,j}\right] \\
    &= \frac{1}{n-h}\sum_{t=h+1}^{n}
       \mathrm{E}\!\left[O_t O_{t-h}\right] \cdot f_{ij}(h) \\
    &= \left(\frac{1}{n-h}\sum_{t=h+1}^{n}
       \mathrm{E}\!\left[O_t O_{t-h}\right]\right) f_{ij}(h),
\end{align*}
where the second equality follows from the independence of $(O_t)$ 
and $(X_t)$, and the third from the stationarity of $(X_t)$. This 
motivates the corrected estimator
\[
    \hat{f}^*_{ij}(h)
    = \frac{\dfrac{1}{n-h}\displaystyle\sum_{t=h+1}^{n} 
      O_t\, O_{t-h}\, Z_{t,i}\, Z_{t-h,j}}
      {\dfrac{1}{n-h}\displaystyle\sum_{t=h+1}^{n} O_t\, O_{t-h}}
    =: \hat{\mathfrak{f}}_{ij}(h) \big/ \overline{O_t O_{t-h}}.
\]
As in Section~\ref{sec:Amp Mod}, the properties of this estimator 
depend on whether $(O_t)$ is deterministic or stochastic. If $(O_t)$ 
is deterministic, then $O_t$ and $O_{t-h}$ are known constants for 
all $t$, and the expectation of the corrected estimator simplifies to
\begin{align*}
    \mathrm{E}\!\left[\hat{f}^*_{ij}(h)\right]
    &= \frac{\dfrac{1}{n-h}\displaystyle\sum_{t=h+1}^{n} 
       O_t\, O_{t-h}\, \mathrm{E}\!\left[Z_{t,i}\, Z_{t-h,j}\right]}
       {\dfrac{1}{n-h}\displaystyle\sum_{t=h+1}^{n} O_t\, O_{t-h}} \\
    &= \frac{\dfrac{1}{n-h}\displaystyle\sum_{t=h+1}^{n} 
       O_t\, O_{t-h}\, f_{ij}(h)}
       {\dfrac{1}{n-h}\displaystyle\sum_{t=h+1}^{n} O_t\, O_{t-h}}
     = f_{ij}(h),
\end{align*}
so the corrected estimator is unbiased in the deterministic case. 
In the stochastic case, however, since numerator and denominator are both 
random, we have in general
\[
    \mathrm{E}\!\left[\frac{\hat{\mathfrak{f}}_{ij}(h)}
    {\overline{O_t O_{t-h}}}\right] \neq 
    \frac{\mathrm{E}\!\left[\hat{\mathfrak{f}}_{ij}(h)\right]}
    {\mathrm{E}\!\left[\overline{O_t O_{t-h}}\right]},
\]
and hence $\hat{f}^*_{ij}(h)$ is generally biased. To derive the asymptotic distribution of $\hat{f}^*_{ij}(h)$, we 
proceed analogously to Section~\ref{sec:Amp Mod} and apply the 
Cram\'er--Wold device to the sample mean of a suitably defined 
binary vector:
\begin{align*}
    \boldsymbol{Z}^{(h)*}_t 
    &= \left(
        Z_{t,-2}^{(h)*},\;
        Z_{t,-1}^{(h)*},\;
        \overbrace{\ldots,\, Z_{t,i}^{(h)*},\, \ldots}^{i=0,\,\ldots,\,m-1},\;
        \overbrace{\ldots,\, Z_{t,m+j}^{(h)*},\, \ldots}^{j=0,\,\ldots,\,m-1}
       \right)^\top\\
    &:= \left(
        O_t O_{t-h},\;
        \underbrace{O_t,\, \ldots,\, O_{t} Z_{t,i},\, \ldots}_{=\,\boldsymbol{Z}^*_t},\;
        \ldots,\, O_t O_{t-h} Z_{t,j} Z_{t-h,j},\, \ldots
        \right)^\top,
\end{align*}
The components of $\boldsymbol{Z}^{(h)^*}_t$ are indexed from $-2$ to $2m-1$: the component at 
index $-2$ captures the product $O_tO_{t-h}$, the components at 
indices $-1$ through $m-1$ coincide with $\boldsymbol{Z}_t^*$ from 
Section~\ref{sec:Amp Mod}, and the remaining components at indices 
$m$ through $2m-1$ capture the lag-$h$ interaction terms 
$O_tO_{t-h}Z_{t,j}Z_{t-h,j}$. Since $\kappa_{\mathrm{ord}}$ depends 
only on the diagonal frequencies $\hat{f}^*_{jj}(h)$, it suffices to 
include the corresponding diagonal components in 
$\boldsymbol{Z}_t^{(h)*}$. Should one wish to study a dependence 
measure that also involves off-diagonal frequencies $\hat{f}^*_{ij}(h)$ 
with $i \neq j$, the vector can straightforwardly be extended by the 
additional components $O_t O_{t-h} Z_{t,i} Z_{t-h,j}$. \\
Since $\boldsymbol{Z}^{(h)*}_t$ is a measurable function of 
$(O_t, O_{t-h}, X_t, X_{t-h})$, it inherits strict stationarity 
and $\alpha$-mixing with exponentially decaying coefficients from 
the underlying processes (by the same argument as in 
Section~\ref{sec:Amp Mod}), and all moment conditions are trivially 
satisfied as its components are bounded indicator variables. 
The Cram\'er--Wold device therefore yields
\[
    \sqrt{n}\!\left(\bar{\boldsymbol{Z}}^{(h)*}_n - 
    \boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}}\right) 
    \xrightarrow{d} \mathcal{N}\!\left(\boldsymbol{0},\, 
    \boldsymbol{\Sigma}_{\boldsymbol{Z}^{(h)*}}\right),
\]
where $\bar{\boldsymbol{Z}}^{(h)*}_n := \frac{1}{n}\sum_{t=1}^{n} 
\boldsymbol{Z}^{(h)*}_t$ and
\[\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}} = E\left[\boldsymbol{Z}^{(h)*}\right] = \left( \pi(h), \pi , \pi \boldsymbol{f}^\top, \pi(h)f_{00}(h), \ldots, \pi(h)f_{m-1m-1}(h)  \right)^\top .\]

To derive the asymptotic null distribution of $\hat{\kappa}(h)$, we 
evaluate $\boldsymbol{\Sigma}_{\boldsymbol{Z}^{(h)*}}$ under the null 
hypothesis that $(X_t)$ is i.i.d. Under this assumption, 
$f_{ij}(h) = f_if_j$ for all $h \geq 1$, which considerably simplifies 
the higher-order moments of $\boldsymbol{Z}^{(h)*}_t$. Before computing 
the individual elements $\sigma^{(h)}_{ij}$ of 
$\boldsymbol{\Sigma}_{\boldsymbol{Z}^{(h)*}}$, we collect the relevant 
higher-order moments of $\boldsymbol{Z}^{(h)*}_t$ under $H_0$, which 
will be used repeatedly in the calculations below:
\begin{align*}
    &\mathrm{E}[Z_{t,i}Z_{t-h,i}Z_{t-k,}Z_{t-k-h,j}]-f_{ii}(h)f_{jj}(h) = \begin{cases}
        f^2_{\mathrm{min}\{i,j\}} -f_i^2f_j^2& \text{ if } k = 0\\
        f_i f_j f_{\min\{i,j\}}-f_i f_j^2 & \text{ if } k-h = 0\\
        0 & \text{else.}
    \end{cases}\\
    &\mathrm{E}[Z_{t+k,i} Z_{t,j}Z_{t-h,j}]-f_i f_{jj}(h) = \begin{cases}
        f_j f_{\min\{i,j\}} -f_i f_j^2 &\text{ if } k = 0 \\
        0 & \text{ else.}
    \end{cases}\\
    &\mathrm{E}[Z_{t,i} Z_{t+k,j}Z_{t+k-h,j}] - f_i f_{jj}(h)
         = \begin{cases}
        f_j f_{\min\{i,j\}} -f_i f_j^2 &\text{ if } k, k-h = 0 \\
        0 & \text{ else.}
    \end{cases}.
\end{align*}
With these simplifications in hand, the individual elements 
$\sigma^{(h)}_{ij}$ of $\boldsymbol{\Sigma}_{\boldsymbol{Z}^{(h)*}}$ 
are calculated as follows:
{\small
\allowdisplaybreaks
\begin{align*}
&\textbf{(i)}\ i = j = -2: \\
&\quad \mathrm{Var}(O_t O_{t-h}) + 2\sum_{k=1}^\infty \mathrm{Cov}(O_t O_{t-h}, O_{t+k}O_{t+k-h}) \\
&\quad = \pi(h)(1-\pi(h)) + 2\sum_{k=1}^{\infty}\bigl(\mathrm{E}[O_t O_{t-h} O_{t+k}O_{t+k-h}]-\pi(h)^2 \bigr) \\[2ex]
%
&\textbf{(ii)}\ i = -2,\ j = -1: \\
&\quad \mathrm{Cov}(O_t O_{t-h}, O_t) 
+ \sum_{k=1}^\infty \bigl[\mathrm{Cov}(O_t O_{t-h}, O_{t+k})
+ \mathrm{Cov}(O_{t+k} O_{t+k-h}, O_t)\bigr] \\
&\quad = (1-\pi)\pi(h) 
+ \sum_{k=1}^{\infty} \bigl[\mathrm{E}[O_t O_{t-h} O_{t+k}] 
+ \mathrm{E}[O_{t+k}O_{t+k-h} O_t] - 2\pi\pi(h)\bigr] \\[2ex]
%
&\textbf{(iii)}\ i = -2,\ j = 0, \ldots, m-1: \\
&\quad \mathrm{Cov}(O_t O_{t-h}, O_t Z_{t,j}) 
+ \sum_{k=1}^{\infty}\bigl[\mathrm{Cov}(O_t O_{t-h}, O_{t+k}Z_{t+k,j}) 
+ \mathrm{Cov}(O_{t+k}O_{t+k-h}, O_t Z_{t,j})\bigr] \\
&\quad = \mathrm{E}[O_t^2 O_{t-h}]\,\mathrm{E}[Z_{t,j}] 
- \mathrm{E}[O_t O_{t-h}]\,\mathrm{E}[O_t]\,\mathrm{E}[Z_{t,j}]\\
&\qquad + \sum_{k=1}^{\infty}\bigl[
\mathrm{E}[O_t O_{t-h} O_{t+k}]\,\mathrm{E}[Z_{t+k,j}]

=: f_j\, \sigma^{(h)}_{-2,-1} \\[2ex]
%
&\textbf{(iv)}\ i = -2,\ j = m, \ldots, 2m-1: \\
&\quad \mathrm{Cov}(O_t O_{t-h},\, O_t O_{t-h}Z_{t,(j-m)}Z_{t-h,(j-m)})\\ 
&\qquad + \sum_{k=1}^{\infty}\bigl[
\mathrm{Cov}(O_t O_{t-h},\, O_{t+k}O_{t+k-h}Z_{t+k,(j-m)}Z_{t+k-h,(j-m)}) \\
&\qquad + \mathrm{Cov}(O_{t+k}O_{t+k-h},\, O_t O_{t-h}Z_{t,(j-m)}Z_{t-h,(j-m)})\bigr] \\
&\quad = \mathrm{E}[O_t^2 O_{t-h}^2]\,\mathrm{E}[Z_{t,(j-m)}Z_{t-h,(j-m)}] 
- \pi(h)^2\, f_{(j-m),(j-m)}(h)\\
&\qquad + \sum_{k=1}^{\infty}\bigl[
\mathrm{E}[O_t O_{t-h} O_{t+k}O_{t+k-h}]\,\mathrm{E}[Z_{t+k,(j-m)}Z_{t+k-h,(j-m)}] \\
&\qquad + \mathrm{E}[O_t O_{t-h} O_{t+k}O_{t+k-h}]\,\mathrm{E}[Z_{t,(j-m)}Z_{t-h,(j-m)}]
- 2\pi(h)^2\, f_{(j-m),(j-m)}(h)\bigr]\\

+ 2\sum_{k=1}^{\infty}\bigl[\mathrm{E}[O_t O_{t-h} O_{t+k}O_{t+k-h}]
- \pi(h)^2\bigr]\Bigr)
=: f_{(j-m),(j-m)}(h)\,\sigma^{(h)}_{-2,-2} \\[2ex]
%
&\textbf{(v)}\ i = -1,\ j = -1: \\
&\quad \mathrm{Cov}(O_t, O_t) + 2\sum_{k=1}^\infty \mathrm{Cov}(O_t, O_{t+k}) 
= \sigma_{\boldsymbol{Z}^*;-1,-1}\\[2ex]
%
&\textbf{(vi)}\ i = -1,\ j = 0, \ldots, m-1: \\
&\quad \mathrm{Cov}(O_t, O_t Z_{t,j}) 
+ \sum_{k=1}^\infty \bigl[\mathrm{Cov}(O_t, O_{t+k}Z_{t+k,j}) 
+ \mathrm{Cov}(O_{t+k}, O_t Z_{t,j})\bigr]
= f_j\,\sigma_{\boldsymbol{Z}^*;-1,-1} \\[2ex]
%

&\quad \mathrm{Cov}(O_t,\, O_t O_{t-h} Z_{t,(j-m)}Z_{t-h,(j-m)}) \\

+ \mathrm{Cov}(O_{t+k},\, O_t O_{t-h}Z_{t,(j-m)}Z_{t-h,(j-m)})\bigr] \\
&\quad = f_{(j-m),(j-m)}(h)\Bigl((1-\pi)\pi(h)
+ \sum_{k=1}^\infty \bigl[\mathrm{E}[O_t O_{t+k} O_{t+k-h}] 
+ \mathrm{E}[O_t O_{t+k} O_{t-h}] - 2\pi\pi(h)\bigr]\Bigr)\\
&\quad = f_{(j-m),(j-m)}(h)\,\sigma^{(h)}_{-2,-1} \\[2ex]

To now obtain the joint distribution of the $\hat{f}^*_{i}$ and $\hat{f}^*_{jj}(h)$ for all $j, i = 0, \ldots,~ m-1$ we define vector valued function $\boldsymbol{g}: [0;1]^{2(m+1)}\rightarrow [0;1]^{2m}$, where
\[g_j(\boldsymbol{x}) = \begin{cases}
    x_j/x_{-1}& \text{ for } j = 0, \ldots, ~ m-1 \\
    x_j/x_{-2}& \text{ for } j = m, \ldots, ~ 2m-1 \\
\end{cases}\]
such that 
\[g_j(\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}}) = f_j \quad \text{and} \quad g_{j+m}(\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}}) = f_{jj}(h) \quad \text{for } j = 0, \ldots, ~m-1  .\]
This allows us to write the corrected estimator $\hat{\boldsymbol{f}}^{(h)*}$ as
\[
    \hat{\boldsymbol{f}}^{(h)*} 
    := \boldsymbol{g}\!\left(\bar{\boldsymbol{Z}}^{(h)*}_n\right)
    = \left(\hat{f}^*_0,\, \ldots,\, \hat{f}^*_{m-1},\, 
      \hat{f}^*_{00}(h),\, \ldots,\, 
      \hat{f}^*_{m-1,m-1}(h)\right)^\top,

with asymptotic mean
\[
    \boldsymbol{\mu}^{(h)*} 
    := \boldsymbol{g}\!\left(\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}}\right)
    = \left(f_0,\, \ldots,\, f_{m-1},\, 
      f_{00}(h),\, \ldots,\, f_{m-1,m-1}(h)\right)^\top.
\]
To apply the delta method, we compute the Jacobian $\boldsymbol{G}$ 
of $\boldsymbol{g}$ evaluated at $\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}}$:

\begin{align*}
    &\boldsymbol{G}_{2m \times 2(m+1)} = \begin{pmatrix}
        \frac{g_0(\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}})}{\delta x_{-2}}& \cdots & \frac{g_0(\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}})}{\delta x_{2m - 1}}\\
        \vdots& \ddots &\vdots \\
        \frac{g_{j+m-1}(\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}})}{\delta x_{-2}}& \cdots & \frac{g_{j+m-1}\boldsymbol{\mu}_{\boldsymbol{Z}^{(h)*}})}{\delta x_{2m - 1}} 
    \end{pmatrix} \\
    &= 
    \begin{pmatrix}
    0 & \frac{-f_0}{\pi} & \frac{1}{\pi}& 0 &\cdots&&\cdots & 0 \\
    \vdots &\vdots & 0 &\ddots & \ddots &&& \vdots \\
    0 & \frac{-f_{m-1}}{\pi}& \vdots & \ddots & \frac{1}{\pi} \\
    \frac{-f_{00}(h)}{\pi(h)} & 0 & & & & \frac{1}{\pi(h)}&\ddots&\vdots\\
    \vdots & \vdots & \vdots & & & \ddots & \ddots&0\\
    \frac{-f_{m-1, m-1}(h)}{\pi(h)} & 0 & 0 & \cdots& &\cdots & 0&\frac{1}{\pi(h)}      
    \end{pmatrix}.
\end{align*}
By the multivariate delta method, the asymptotic covariance matrix of $\sqrt{n}(\hat{\boldsymbol{f}}^{(h)*}- \mu)$ is given by $\boldsymbol{\Sigma}_{\boldsymbol{f}^{(h)*}} = \boldsymbol{G}\boldsymbol{\Sigma}_{\boldsymbol{Z}^{(h)*}} \boldsymbol{G}^\top$. Carrying out the matrix multiplication row by row yields:
\begin{align*}
    \boldsymbol{G}\boldsymbol{\Sigma}_{\boldsymbol{Z}^{(h)*}} \boldsymbol{G}^\top &= \begin{pmatrix}
        \frac{- f_0}{\pi} \sigma^{(h)}_{-1, -2} + \frac{1}{\pi} \sigma^{(h)}_{0, -2} & \cdots & \frac{-f_0}{\pi} \sigma^{(h)}_{-1, 2m-1} +  \frac{1}{\pi} \sigma^{(h)}_{0, 2m-1} \\
        \vdots &\ddots & \vdots \\
        \frac{-f_{m-1}}{\pi} \sigma^{(h)}_{-1, -2} + \frac{1}{\pi} \sigma^{(h)}_{m-1, -2} & \cdots & \frac{-f_{m-1}}{\pi} \sigma^{(h)}_{-1, 2m-1} +  \frac{1}{\pi} \sigma^{(h)}_{m-1, 2m-1} \\
        \frac{-f_{00}(h)}{\pi(h)} \sigma^{(h)}_{-2, -2} + \frac{1}{\pi(h)} \sigma^{(h)}_{m, -2} & \cdots & \frac{-f_{00}(h)}{\pi(h)} \sigma^{(h)}_{-2, 2m-1} +  \frac{1}{\pi(h)} \sigma^{(h)}_{m, 2m-1} \\
        \vdots &\ddots& \vdots \\
        \frac{f_{m-1}(h)}{\pi(h)} \sigma^{(h)}_{-2, -2} + \frac{1}{\pi(h)} \sigma^{(h)}_{2m-1, -2} & \cdots & \frac{-f_{m-1, m-1}(h)}{\pi(h)} \sigma^{(h)}_{-2, 2m-1} +  \frac{1}{\pi(h)} \sigma^{(h)}_{2m-1, 2m-1}

\end{align*}

\begin{align*}
    \mathrm{Bias}(\hat{\kappa}_{\mathrm{ord}})
    &= \frac{1}{2}\mathrm{tr}(\boldsymbol{H}_{\kappa}\boldsymbol{\Sigma}_{\boldsymbol{f}^{(h)*}})
    = \frac{1}{2}\sum_{i=0}^{2m-1}\left[\boldsymbol{H}_{\kappa}
    \boldsymbol{\Sigma}_{\boldsymbol{f}^{(h)*}}\right]_{ii} \\
    &= \frac{1}{2} \sum_{i=0}^{m-1}\Bigl[2\Bigl(-\frac{1}{\sum_{k=0}^{m-1}f_k(1-f_k)}\sigma_{i,i}^{(h)*} + \sum_{l=0}^{m-1} 
        \frac{f_i + f_l - 4f_i f_l}
             {\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2}\sigma_{l,i}^{(h)*}\Bigr) \\
    &\qquad + \sum_{l=m}^{2m-1}  
        \frac{-(1-2f_i)}
             {\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2}\, 
        \sigma_{l,i}^{(h)*}\Bigr] + \frac{1}{2} \sum_{i=m}^{2m-1}\Bigl[\sum_{l=0}^{m-1}  
        \frac{-(1-2f_i)}
             {\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2}\, 
        \sigma_{i,l}^{(h)*}\Bigr]\\
    &= -\frac{\sum_{i=0}^{m-1}\sigma_{ii}^{(h*)}}{\sum_{k=0}^{m-1}f_k(1-f_k)} + \frac{\sum_{i,l = 0}^{m-1}(f_i + f_l - 4f_i f_l)\sigma_{l,i}^{(h)*}}
             {\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2} 
        - \frac{\sum_{i,l=0}^{m-1}(1-2f_i)\sigma_{i, m+l}^{(h)*}}{\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2} \\
    &= -\frac{\frac{1}{\pi}\sum_{i=0}^{m-1}(f_i- f_i^2)}{\sum_{k=0}^{m-1}f_k(1-f_k)} +  \frac{\sum_{i,l = 0}^{m-1}(f_i + f_l - 4f_i f_l)\sigma_{l,i}^{(h)*}- \sum_{i,l=0}^{m-1}(1-2f_i)2f_l\sigma^{(h)*}_{l,i}}{\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2} \\
    &= -\frac{1}{\pi} + \frac{\sum_{i,l = 0}^{m-1}\left(f_i + f_l - 4f_i f_l - 2f_l+ 4 f_j f_l \right)\sigma^{(h)*}_{l,i}}{\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2}
    = -\frac{1}{\pi} + \frac{\overbrace{\sum_{i,l = 0}^{m-1}\left(f_i - f_l \right)}^{=0}\sigma^{(h)*}_{l,i}}{\left(\sum_{k=0}^{m-1}f_k(1-f_k)\right)^2} \\
    &= -\frac{1}{\pi}
\end{align*}
Note that we substituted $\sigma^{(h)*}_{i, m+l} = 2f_l\sigma_{i,l}^{(h*)}$ from case \textbf{(ii)} of $\boldsymbol{\Sigma}_{\boldsymbol{f}^{(h)*}}$ and used its symmetry by substituting $\sigma_{i,l}^{(h)*}= \sigma_{l,i}^{(h)*}$. \\
The results of the previous derivations are summarized in the following theorem
\begin{theorem}[Asymptotic distribution and bias of $\hat{\kappa}_{\mathrm{ord}}(h)$]
\label{thm:kappa-asymp}
Let the assumptions of theorem \ref{thm:clt_fstarhat} hold and denote
\[\pi(h_1, \ldots, h_r):= \mathrm{E}[O_0\cdot O_{h_1}\ldots O_{h_r}].\] Then, under the null hypothesis $H_0$ 
that $(X_t)$ is i.i.d., the following hold as $n \to \infty$:

\begin{enumerate}
    \item \textbf{Asymptotic normality.} The estimator 
    $\hat{\kappa}_{\mathrm{ord}}(h)$ satisfies
    \[
        \sqrt{n}\,\hat{\kappa}_{\mathrm{ord}}(h) 
        \xrightarrow{d} \mathcal{N}(0, \sigma^2_\kappa),
    \]
    where
    \[
        \sigma^2_\kappa =
        \frac{
            \frac{1}{\pi(h)} \sum_{i,j=0}^{m-1}
            \bigl(f_{\min\{i,j\}} - f_i f_j\bigr)
            \Bigl(
                f_{\min\{i,j\}} 
                + \Bigl(
                    1 + \frac{2\pi(h,2h)}{\pi(h)} 
                    - \frac{4\pi(h)}{\pi}
                \Bigr) f_i f_j
            \Bigr)
        }{
            \left(
                \sum_{k=0}^{m-1} f_k (1 - f_k)
            \right)^2
        }.
    \]

    \item \textbf{Finite-sample bias.} The approximate bias of 
    $n \hat{\kappa}_{\mathrm{ord}}(h)$ under $H_0$ is
    \[
        \mathrm{Bias}\bigl(n \hat{\kappa}_{\mathrm{ord}}(h)\bigr) 
        \approx -\frac{1}{\pi},
    \]
    so that the bias-corrected estimator is given by
    \[
        \hat{\kappa}^{\mathrm{bc}}_{\mathrm{ord}}(h) 
        := \hat{\kappa}_{\mathrm{ord}}(h) + \frac{1}{n\pi}.
    \]
\end{enumerate}
\end{theorem}
